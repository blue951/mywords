ajax和promise的结合使用
        在需要依赖完成的ajax请求可使用promise保证执行顺序 在第一个请求正确返回后再发送第二个请求

        /*
        定义一个使用promise的ajax请求,这里依赖jquery
        参数中请求url为必填参数
        */
        const ajaxPromise=  param => {
        return new Promise((resovle, reject) => {
            $.ajax({
            "type":param.type || "get",
            "async":param.async || true,
            "url":param.url,
            "data":param.data || "",
            "success": res => {
                resovle(res);
            },
            "error": err => {
                reject(err);
            }
            })
        })
        }
        
        /*
        第一个请求
        */
        let step1 = () => {
            ajaxPromise({
            "url":"",
            }).then(res => {
                console.log("第一个请求正确返回==>"+res); 
                step2(res); 
            }).catch(err => {
                console.log("第一个请求失败"); 
            })
        }
        
        /*
        第二个请求
        */
        let step2 = (res) => {
            ajaxPromise({
            "type":"get",
            "url":"",
            "data":{"name":res}
            }).then(res => {
                console.log("第二个请求正确返回==>"+res); 
            }).catch(err => {
                console.log("第二个请求失败==>"+err); 
            })
        }
        
        step1();

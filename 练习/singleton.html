function SingleTon () {
            this.instance = null;
        }

        SingleTon.getInstance = function(){
            if(!this.instance){
                this.instance = new SingleTon();
            }
            return this.instance;
        }
        
        SingleTon.prototype.getName = function(){
            console.log('getName');
        }

        var objOne = SingleTon.getInstance();
        var objTwo = SingleTon.getInstance();
        console.log(objOne === objTwo);  //true
        console.log((new SingleTon()).getName()); //getName

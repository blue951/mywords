function before(str){
            return "before" + str;
        };

        function after(str){
            return str + "after";
        };

        function compose(){
            var fns = [].slice.call(arguments);
            return function(param){
                var res = param;
                for(var i=0; i < fns.length; i++){
                    res = fns[i](res);
                }
                return res;
            }
        };

        var demoFn = compose(before,after);
        console.log(demoFn("fwf"));
